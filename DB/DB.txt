CREATE DATABASE WKWKDISCORD;

USE WKWKDISCORD;

CREATE TABLE BOTDATA(
 BOTTOKEN CHAR(59) PRIMARY KEY NOT NULL,
 BOTCLIENTID VARCHAR(20) NOT NULL,
 CLIENTSECRET VARCHAR(40) NOT NULL
);

CREATE TABLE SERVERPROPERTY (
 SERVERID CHAR(25) PRIMARY KEY NOT NULL,
 MENTIONCHANNELID VARCHAR(30),
 FIRSTCHANNELID VARCHAR(30),
 VOICECATEGORYID VARCHAR(30),
 TEXTCATEGORYID VARCHAR(30),
 PREFIX VARCHAR(100) NOT NULL,
 TEMPBY VARCHAR(1) DEFAULT '1',
 TEXTBY VARCHAR(1) DEFAULT '1',
 DEFAULTSIZE CHAR(3) DEFAULT '0'
);

CREATE TABLE TEMPCHANNELS (
 VOICECHANNELID VARCHAR(30) PRIMARY KEY NOT NULL,
 TEXTCHANNELID VARCHAR(30) NOT NULL,
 SERVERID VARCHAR(30) NOT NULL,
 HIDEBY TINYINT(1) DEFAULT 0,
 LOCKBY TINYINT(1) DEFAULT 0
);

CREATE TABLE MENTIONMESSAGES (
 SERVERID VARCHAR(30) NOT NULL,
 MESSAGEID VARCHAR(30) NOT NULL,
 TEXTCHANNELID VARCHAR(30) NOT NULL,
 PRIMARY KEY (TEXTCHANNELID,MESSAGEID)
);

CREATE TABLE REACTMESSAGE (
 SERVERID VARCHAR(30) PRIMARY KEY NOT NULL,
 TEXTCHANNELID VARCHAR(30) NOT NULL,
 MESSAGEID VARCHAR(30) NOT NULL
);

CREATE TABLE REACTROLES (
 MESSAGEID VARCHAR(30) NOT NULL,
 ROLEID VARCHAR(30) NOT NULL,
 EMOJI VARCHAR(30) NOT NULL,
 PRIMARY KEY (MESSAGEID,EMOJI)
);

CREATE TABLE NAMEPRESET (
 SERVERID VARCHAR(30) NOT NULL,
 NAME VARCHAR(100) NOT NULL,
 PRIMARY KEY (SERVERID,NAME)
);
